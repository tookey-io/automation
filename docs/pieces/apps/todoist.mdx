## Actions


      <Card title="Create Task " >
        Create task

        <Expandable title="Properties">
          
<ResponseField name="authentication" required type="Oauth2">

</ResponseField>
            
<ResponseField name="Project"  type="Dropdown">
Task project ID. If not set, task is put to user's Inbox.
</ResponseField>
            
<ResponseField name="content" required type="Long_text">
The task's content. It may contain some markdown-formatted text and hyperlinks
</ResponseField>
            
<ResponseField name="Description"  type="Long_text">
A description for the task. This value may contain some markdown-formatted text and hyperlinks.
</ResponseField>
            
<ResponseField name="Labels"  type="Array">
The task's labels (a list of names that may represent either personal or shared labels)
</ResponseField>
            
<ResponseField name="Priority"  type="Number">
Task priority from 1 (normal) to 4 (urgent)
</ResponseField>
            
<ResponseField name="Due date"  type="Short_text">
Specific date in YYYY-MM-DD format relative to user's timezone
</ResponseField>
              
        </Expandable>
      </Card>
